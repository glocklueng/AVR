35 void avr_spi_begin(void)
36 {
37  uint8_t value = DDRB;
38  value |= _BV(SPI_MOSI_PIN); // output
39  value &= ~_BV(SPI_MISO_PIN);// input
40  value |= _BV(SPI_SCK_PIN); // output
41  value |= _BV(SPI_SS_PIN); // output
42  DDRB = value;
43 
44  PORTB |= _BV(SPI_SS_PIN); // set SS to high
45 
46  SPCR = _BV(SPE) | _BV(MSTR) | _BV(SPR1) | _BV(SPR0);
47 }
